CREATE SEQUENCE RMORDER_CODE

CREATE TABLE ORDER_RAWMATERIAL(
RMORDER_CODE VARCHAR2(100) PRIMARY KEY,
MMCODE VARCHAR2(100) REFERENCES RAWMATERIAL(RAWM_CODE),
SUPPLIERSEQ VARCHAR2(100) REFERENCES RAWMATERIAL(SUPPLIERSEQ),
RMORDER_DATE DATE,
RMORDER_AMOUNT NUMBER(37),
RMORDER_STATUS VARCHAR2(100),
WAREHOUSE_CODE VARCHAR2(100),
RMORDER_REQUSERID VARCHAR2(100),
RMORDER_REQDATE DATE,
MFSEQ VARCHAR2(100) REFERENCES MANUFACTURE(MFSEQ)
)
