CREATE SEQUENCE RAWM_CODE

CREATE TABLE RAWMATERIAL(
RAWM_CODE VARCHAR2(50) PRIMARY KEY,
SUPPLIERSEQ VARCHAR2(50) REFERENCES SUPPLIER(SUPPLIERSEQ),
RAWM_NAME VARCHAR2(50),
RAWM_SORT VARCHAR2(50),
RAWM_PRICE NUMBER(37),
RAWM_MINIMUM NUMBER(37),
RAWM_SET NUMBER(37),
RAWM_AUTOORDER VARCHAR2(50) CHECK(RAWM_AUTOORDER IN('CHECKED','NOCHECK')),
RAWM_DELETE VARCHAR2(50)
)
